Browser

Install​

$ npm install @rmbg/browser

Usage​

import rmbg from '@rmbg/browser'
import { createSiluetaModel } from '@rmbg/browser/models'

rmbg(source, {
  model: createSiluetaModel(),
  onProgress(progress) {
    console.log(progress)
  }
})
  .then((image) => {
    console.log(image)
  })
  .catch((err) => {
    console.error(err)
  })
